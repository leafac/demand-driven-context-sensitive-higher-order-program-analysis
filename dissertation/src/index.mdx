import { Helmet } from "react-helmet";
import { MDXProvider } from "@mdx-js/react";
import GitHubSlugger from "github-slugger";

export default class extends React.Component {
  constructor(props) {
    super(props);
    const global = this;
    this.state = { tableOfContents: [] };
    this.slugger = new GitHubSlugger();
    this.numberer = new (class {
      constructor() {
        this.counter = [];
      }
      number(level) {
        this.counter.splice(level);
        while (this.counter.length < level) this.counter.push(0);
        this.counter[this.counter.length - 1]++;
        return this.counter.join(".");
      }
    })();
    this.components = {
      style: ({ children }) =>
        React.createElement("style", {
          dangerouslySetInnerHTML: { __html: children },
        }),
      ...Object.fromEntries(
        [1, 2, 3, 4, 5, 6].map((level) => {
          const type = `h${level}`;
          return [
            type,
            class extends React.Component {
              constructor(props) {
                super(props);
                this.state = { id: null, heading: null };
              }
              componentDidMount() {
                const id = global.slugger.slug(this.props.children);
                const number = global.numberer.number(level);
                const heading = (
                  <>
                    <span class="heading-number">{number} </span>
                    {this.props.children}
                    <code className="draft"> (#{id})</code>
                  </>
                );
                this.setState({ id, heading });
                global.setState((state) => ({
                  tableOfContents: [...state.tableOfContents, { id, heading }],
                }));
              }
              render() {
                return React.createElement(
                  type,
                  { id: this.state.id },
                  this.state.heading
                );
              }
            },
          ];
        })
      ),
      TableOfContents: () => (
        <ul className="table-of-contents">
          {global.state.tableOfContents.map(({ id, heading }) => (
            <li key={id}>
              <a href={`#${id}`}>{heading}</a>
            </li>
          ))}
        </ul>
      ),
    };
  }
  render() {
    return (
      <MDXProvider components={this.components}>
        {this.props.children}
      </MDXProvider>
    );
  }
}
